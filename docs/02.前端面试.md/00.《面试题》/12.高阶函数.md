## 函数柯里化

```js
function add(x) {

    return function(y) {

        return x +y;
    }
}

let sum = add(3);
console.log(sum(2))

// or


function add(x) {

    return function(y) {

        return x +y;
    }
}
console.log(add(2)(3))
```
